\section[NFA]{Non-deterministiske endelige automater}

\begin{frame}[t]
  \hfill\xymatrix{%
    \ar[r] & *+<2ex>[o][F]{q_1} \ar[r]^1 \ar@(ul,ur)[]^(.65){0,1}
    & *+<2ex>[o][F]{q_2} \ar[r]^{0, \epsilon} 
    & *+<2ex>[o][F]{q_3} \ar[r]^1
    & *+<2ex>[o][F=]{q_4} \ar@(ul,ur)[]^(.65){0,1}
  }\hfill\mbox{}

  \bigskip \alert{Definition 1.37:} En \alert{nondeterministisk endelig
    automat} (\alert{NFA}) er en 5-tupel \alert{$M= (Q, \Sigma, \delta,
    q_0, F)$}, hvor delene er
  \begin{enumerate}
  \item $Q$ : en endelig mængde af tilstande
  \item $\Sigma$ : input-alfabetet
  \item \alert{$\delta: Q\times\big( \Sigma\cup\{ \epsilon\}\big)\to
      \Pot(Q)$} : transitions-funktionen
  \item $q_0\in Q$ : starttilstanden
  \item $F\subseteq Q$ : mængden af accepttilstande
  \end{enumerate}

  \medskip $M$ siges at \alert{acceptere} et ord $w\in \Sigma^*$ hvis der
  findes $m\in \Nat$ og $y_1, y_2,\dots, y_m\in \Sigma\cup\{ \epsilon\}$
  og $r_0, r_1,\dots, r_m\in Q$ således at $w= y_1 y_2\dots y_m$ og
  \begin{enumerate}
  \item $r_0= q_0$,
  \item $r_{i+1}\in \delta( r_i, y_{i+1})$ for alle $i= 0, 1,\dots, m-1$, og
  \item $r_m\in F$.
  \end{enumerate}
\end{frame}

\begin{frame}
  \begin{itemize}
  \item enhver DFA er også en NFA
  \item enhver NFA kan laves om til en DFA der genkender samme sprog
    (\alert{delmængdekonstruktionen})
  \item et sprog er defineret til at være \alert{regulært} hvis der er en
    DFA der genkender det
  \item[\IMPL] et sprog er regulært hvis og kun hvis der er en NFA der
    genkender det
  \item regulære sprog er lukket under $\cup$, $\circ$, ${}^*$ (vises ved
    at konstruere en ny NFA ud fra de givne NFAs)
  \item regulære sprog er lukket under $\cap$ og $\bar{~}$
    (komplement) (vises ved at konstruere en ny \alert DFA ud fra de givne
    \alert DFAs; konstruktionerne virker \emph{kun\/} for DFAs!)
  \item NFAs er generelt mere simple at fremstille
  \item men nogen gange kan det være nødvendigt at arbejde med DFAs --
    eksempel: opgave 1.13
  \end{itemize}
\end{frame}

\section[NFA vs.\ RE]{NFAs og regulære udtryk}

\begin{frame}
  \alert{Lemma 1.55:} Hvis et sprog beskrives ved et regulært udtryk, da
  er det regulært.

  \medskip \alert{Bevises} ved \alert{strukturel induktion}:
  \begin{enumerate}
  \item konvertér de basale regulære udtryk til NFAs
  \item brug lukningsegenskaber til at konvertere sammensætninger af
    regulære udtryk til sammensætninger af NFAs
  \item \alert{Smart!}
  \end{enumerate}

  \bigskip I dag: \alert{Lemma 1.60:} Hvis et sprog er regulært, da kan
  det beskrives ved et regulært udtryk.

  \medskip (Bevises ved at \emph{generalisere\/} NFAs til \alert{GNFA}s.)

  \bigskip \IMPL\ \alert{Sætning 1.54:} Et sprog er regulært \blue{hvis og
    kun hvis} det kan beskrives ved et regulært udtryk.
\end{frame}

\section[NFA $\Rightarrow$ RE]{Regulære sprog genereres af regulære udtryk}

\begin{frame}[t]
  \alert{Lemma 1.60:} Givet et alfabet $\Sigma$ og et regulært sprog
  $L\subseteq \Sigma^*$, da findes et regulært udtryk $R$ over $\Sigma$
  således at $L= \llbracket R\rrbracket$.

  \medskip \alert{Nøgle} til beviset: Ny slags maskiner der kombinerer NFA
  og regulære udtryk: \alert{generaliserede nondeterministiske endelige
    automater} (\alert{GNFA})

  \bigskip \alert{Definition 1.64:} En \alert{GNFA} er en 5-tupel
  \alert{$(Q, \Sigma, \delta, q_0, q_f)$}, hvor delene er
  \begin{enumerate}
  \item $Q$ : en endelig mængde af tilstande
  \item $\Sigma$ : input-alfabetet
  \item \alert{$\delta:\big( Q\setminus\{ q_f\}\big)\times\big(
      Q\setminus\{ q_0\}\big)\to \mathcal R$} : transitions-funktionen
  \item $q_0\in Q$ : starttilstanden
  \item \alert{$q_f\in Q$} : accepttilstanden
  \end{enumerate}

  \medskip \alert{Notation:} $\alert{\mathcal R}= \mathcal R( \Sigma)=$
  mængden af alle \alert{regulære udtryk} over et givet alfabet $\Sigma$.

  \medskip (Bemærk at GNFAs introduceres kun for det her bevis. De bruges
  ikke til andet.)
\end{frame}

\begin{frame}[t]
  Definition 1.64: En \alert{GNFA} er en 5-tupel $(Q, \Sigma,
    \delta, q_0, q_f)$, hvor delene er
  \begin{enumerate}
    \setcounter{enumi}2
  \item $\delta:\big( Q\setminus\{ q_f\}\big)\times\big(
      Q\setminus\{ q_0\}\big)\to \mathcal R$ : transitions-funktionen
    \setcounter{enumi}4
  \item $q_f\in Q$ : accepttilstanden
  \end{enumerate}

  \medskip Ligesom NFAs, men
  \begin{itemize}
  \item med kun \alert{én} accepttilstand
  \item med \alert{regulære udtryk} på transitionerne i stedet for tegn
  \item med transitioner \alert{fra enhver tilstand til enhver tilstand}
    (også sig selv), bortset fra at
    \begin{itemize}
    \item starttilstanden ikke har \alert{indgående} transitioner, og at
    \item accepttilstanden ikke har \alert{udgående} transitioner
    \end{itemize}
  \end{itemize}

\begin{frame}[t]
  Definition 1.64: En \alert{GNFA} er en 5-tupel \alert{$(Q, \Sigma,
    \delta, q_0, q_f)$}, hvor delene er
  \begin{enumerate}
  \item $Q$ : en endelig mængde af tilstande
  \item $\Sigma$ : input-alfabetet
  \item $\delta:\big( Q\setminus\{ q_f\}\big)\times\big(
      Q\setminus\{ q_0\}\big)\to \mathcal R$ : transitions-funktionen
  \item $q_0\in Q$ : starttilstanden
  \item $q_f\in Q$ : accepttilstanden
  \end{enumerate}

  \medskip GNFAen \alert{accepterer} et ord $w\in \Sigma^*$ hvis der
  findes $m\in \Nat$ og $y_1, y_2,\dots, y_m\in \Sigma^{\alert *}$
  (\alert!) og $r_0, r_1,\dots, r_m\in Q$ således at $w= y_1 y_2\dots y_m$
  og
  \begin{enumerate}
  \item $r_0= q_0$,
  \item $y_{i+1}\in \llbracket \delta( r_i, r_{i+1})\rrbracket$ for alle
    $i= 0, 1,\dots, m-1$, og
  \item $r_m= q_f$.
  \end{enumerate}

  \bigskip \alert{Bevisidé:} konvertér en DFA til en GNFA og så GNFAen til
  et regulært udtryk ved at \alert{fjerne én tilstand ad gangen}.
\end{frame}

\section{Ikke-regulære sprog}

\begin{frame}
  \emph{Ikke alle sprog er regulære.\/} F.x.\ sproget \alert{$\{ 0^n
    1^n\mid n\in \Nat\}$}:
  \begin{equation*}
    \xymatrix{%
      \ar[r] & *+<2ex>[o][F=]{} \ar[r]^0
      & *+<2ex>[o][F]{} \ar[r]^0 \ar[d]^1
      & *+<2ex>[o][F]{} \ar[r]^0 \ar[d]^1
      && {\cdots} & \ar[r]^0
      & *+<2ex>[o][F]{} \ar[r]^0 \ar[d]^1
      && {\cdots}
      \\ && *+<2ex>[o][F=]{}
      & *+<2ex>[o][F]{} \ar[l]^1
      & \ar[l]^1
      & {\cdots}
      && *+<2ex>[o][F]{} \ar[l]^1
      & \ar[l]^1
      & {\cdots}
    }
  \end{equation*}

  -- en \alert{uendelig} automat!

  \bigskip \emph{Pumping Lemma}: en egenskab ved alle regulære sprog.

  \medskip \IMPL\ Hvis et sprog \emph{ikke\/} har den egenskab, \alert{kan
    det ikke være regulært}.
\end{frame}

\begin{frame}[t]
  \alert{Sætning 1.70 (Pumpelemmaet):} For ethvert regulært sprog $A$
  findes der et (naturligt) tal $p$ således at ethvert ord $s\in A$ der
  har længde mindst $p$ kan opsplittes i tre stykker, $s= xyz$, med
  \begin{itemize}
  \item $|y|> 0$ og $|xy|\le p$,
  \item og således at ordene $x y^i z\in A$ for alle $i\in \Nat_0$.
  \end{itemize}

  \only<1>{%
    \bigskip \emph{En gang til:}\\
    For ethvert regulært sprog $A$\\
    \qquad findes $p\in \Nat_0$ således at\\
    \qquad\qquad for ethvert $s\in A$ med $|s|\ge p$\\
    \qquad\qquad\qquad findes en opsplitning $s= xyz$ således at\\
    \qquad\qquad\qquad\qquad $|y|> 0$ og $|xy|\le p$ og\\
    \qquad\qquad\qquad\qquad for alle $i\in \Nat_0$\\
    \qquad\qquad\qquad\qquad\qquad $x y^i z\in A$.

  \vspace*{20ex}}%
  \only<2>{%
    \bigskip \alert{Eksempel 1.73:} Sproget $B= \{ 0^n 1^n\mid
    n\in \Nat\}$ er ikke regulært.

    \medskip \alert{Bevis} (ved modstrid; kortere end i bogen!):
    \emph{Antag\/} at $B$ er regulært, og lad $p$ være pumpelængden. Lad
    \alert{$s= 0^p 1^p$}, da er $|s|\ge p$.

    \smallskip Lad $s= xyz$ være en opsplitning af $s$ som opfylder
    pumpelemmaets betingelser. Pga.\ \alert{$|xy|\le p$} kan $y$ kun
    indeholde $0$er, og pga.\ \alert{$|y|> 0$} indeholder $y$ mindst ét
    $0$.

    \smallskip Sidste betingelse i lemmaet siger bl.a.\ at ordet
    $\alert{xyyz}\in A$, men dette ord indeholder for mange
    $0$er. \alert{Modstrid!}

  \vspace*{20ex}}%
  \only<3->{%
    \bigskip \alert{Bevis:} Lad $M=( Q, \Sigma, \delta, q_0, F)$ være en
    DFA der genkender $A$, og lad \alert{$p=| Q|$}. Lad $s= s_1 s_2\dots
    s_n\in A$ med $|s|\ge p$.}

  \only<3>{%
    {\color{blue} \smallskip Mens $M$ læser $s$, kommer den igennem en
      følge af $n+1$ tilstande. Men \alert{$n+1> p$}, så der er flere
      tilstande i følgen end der er i $M$!

      \smallskip Dvs.\ der er en tilstand der optræder \alert{to gange} i
      følgen -- en \alert{løkke}!

      \smallskip Hvis vi tager $x$ til at være den del af $s$ der læses
      \emph{før\/} løkken, $y$ den del der læses \emph{i\/} løkken, og $z$
      den del der læses \emph{efter\/} løkken, kan vi gennemløbe løkken
      \alert{$i$} gange og genkende strengen $x y^{\alert i} z$.}

    \vspace*{20ex}}%
  \only<4->{%
    \smallskip Lad $r_1, r_2,\dots, r_{n+1}\in Q$ således at $r_1= q_0$,
    $r_{n+1}\in F$, og $r_{i+1}= \delta( r_i, s_i)$ for alle $i$.

    \smallskip Vi har $n+1\ge p+1$, og $|Q|= p$. Derfor findes indices $j$
    og $\ell$ således at $1\le j< \ell\le p+1$ og $r_j= r_\ell$.

    \smallskip Lad $x= s_1\dots s_{j-1}$, $y= s_j\dots s_{\ell-1}$, $z=
    s_\ell\dots s_n$. Pga.\ $j< \ell$ har vi $|y|\ge 0$, og $\ell\le p+1$
    medfører $|xy|\le p$.

    \smallskip Eftersom $\delta( r_{\ell-1}, s_{\ell-1})= r_j$, er enhver
    følge $(r_1,\dots, r_{j-1})( r_j,\dots, r_{\ell-1})^{\alert i}(
    r_\ell,\dots, r_{n+1})$ en accepterende følge for $M$, og ordet den
    genkender er $x y^{\alert i} z$.

  \vspace*{20ex}}
\end{frame}
