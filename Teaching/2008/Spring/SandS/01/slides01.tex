\section[Regulære udtryk]{Motivation: Regulære udtryk}

\begin{frame}[fragile]
  \alert{Regulære udtryk} bruges til tekstbehandling:
  \begin{itemize}
  \item at \alert{søge} efter mønstre
  \item at \alert{erstatte}
  \end{itemize}

  \medskip Eksempler:
  \begin{itemize}
  \item \verb!grep 'Hans' manual.tex!
  \item \verb!grep 'vi[ ,.]' manual.tex!
  \item \verb!grep 'o[a-zA-Z]*o[a-zA-Z]*o' manual.tex!
    \medskip
  \item \verb!sed 's:\\[a-zA-Z]*: :g' manual.tex!
  \item \verb!sed 's:\\usepackage{[a-zA-Z]*}: :g' manual.tex!
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  Et regulært udtryk definerer et \alert{sprog}:
  \begin{itemize}
  \item $\llbracket \verb!Hans!\rrbracket=\{ \verb!Hans!\}$
  \item $\llbracket \verb!vi[ ,.]!\rrbracket=\{ \verb!vi !,
    \verb!vi,!, \verb!vi.!\}$
  \item $\llbracket \verb!o[a-zA-Z]*o!\rrbracket=\{ \verb!otto!,
    \verb!othello!, \verb!ohMyGodNo!, \verb!oo!, \dots\}$
  \item $\llbracket \verb!\\[a-zA-Z]*!\rrbracket=\{ \verb!\section!,
    \verb!\emph!, \verb!\prut!, \verb!\!, \dots\}$
  \item $\llbracket
    \verb!\\usepackage\(\[[a-zA-Z]*]\)\?{[a-zA-Z]*}!\rrbracket$ $=\{
    \verb!\usepackage{url}!, \verb!\usepackage[danish]{babel}!,$
    $\verb!\usepackage{}!, \verb!\usepackage[]{}!, \dots\}$
  \end{itemize}

  \medskip \alert{Mål} for i dag: At gøre det her \emph{præcist}
\end{frame}


\section{Bogstaver, ord}

\begin{frame}
  \begin{itemize}
  \item $\Sigma$ \quad-- en endelig mængde af \alert{bogstaver} eller
    \alert{symboler}

    -- et \alert{alfabet}
  \item et \alert{ord}: en endelig følge af bogstaver

    -- normalt skrevet uden parenteser eller kommaer
  \item eksempel: $\Sigma= \{ 0, 1\}$
    
    ord over $\Sigma$: f.x.\ $0, 1, 00, 01, 1001010110101$
  \item eksempel: $\Sigma= \{ a, b, c, d, r\}$

    ord over $\Sigma$: f.x.\ $a, b, c, d, r, abba, abracadabra$
  \item eksempel: $\Sigma= \{ \text{gik}, \text{jeg}, \text{land},
    \text{mig}, \text{og}, \text{over}, \text{sø}, \text{to}, \text{vi}\}$

    ord over $\Sigma$: f.x.\ ``jeg og mig og vi to''\\ eller ``jeg gik mig
    over sø og land''
  \item eksempel: $\Sigma= \{ \text{else}, \text{if}, \text{then},
    \text{\textit{Exp}}, \text{\textit{Stm}}\}$

    ord over $\Sigma$: f.x.\ ``if \textit{Exp} then \textit{Stm} else \textit{Stm}''
  \end{itemize}
\end{frame}

\section{Sprog}

\begin{frame}
  \begin{itemize}
  \item et \alert{sprog}: en mængde af ord (endelig eller uendelig)
  \item mængden af \emph{alle} ord over et alfabet $\Sigma$ skrives
    \alert{$\Sigma^*$}

    (den er altid uendelig (medmindre $\Sigma$ er tom \dots))
  \item eksempel: $\Sigma= \{ 0, 1\}$

    $\Sigma^*= \{ \epsilon, 0, 1, 00, 01, 10, 11, 000, 001, 010,\dots \}$
  \item \alert{$\epsilon$} \quad-- det \emph{tomme} ord; ordet af længde 0
  \item \alert{længden} af et ord: $|w|=$ antallet af bogstaver i ordet
  \item \alert{$\emptyset$} \quad-- det \emph{tomme sprog}; mængden uden
    indhold
  \item \alert{Bemærk:} $\epsilon$ er et \emph{ord}, $\emptyset$ er et
    \emph{sprog}. Og \alert{$\{ \epsilon\}\ne \emptyset$}
  \end{itemize}
\end{frame}

\section{At sammensætte ord}

\begin{frame}
  \begin{itemize}
  \item at \alert{sammensætte} ord: $abe\circ kat= abekat$

    (svarer til at gange tal sammen, men \emph{ikke kommutativt!})

    ($\circ$-tegnet udelades de fleste gange)
  \item $\epsilon$ er \alert{identiteten}: $w\circ \epsilon= w$ og
    $\epsilon\circ w= w$ for alle ord $w$.

    (ligesom tallet 1 er identiteten for multiplikation)
  \item gentagen sammensættelse skrives som \alert{potenser}: $a^2= aa$,
    $a^3= aaa$, $a^9= aaaaaaaaa$ etc.
  \end{itemize}
\end{frame}

\section{Operationer på sprog}

\begin{frame}
  Hvis $L_1$ og $L_2$ er sprog over et alfabet $\Sigma$, kan vi danne
  \begin{itemize}
  \item foreningsmængden $\alert{L_1\cup L_2}= \{ w\mid w\in L_1\text{
      eller } w\in L_2\}$

    -- sproget med alle de ord der er i $L_1$ \alert{eller} $L_2$
  \item fællesmængden $\alert{L_1\cap L_2}= \{ w\mid w\in L_1\text{
      og } w\in L_2\}$

    -- sproget med alle de ord der er i $L_1$ \alert{og} $L_2$
  \item sammensætningen $\alert{L_1\circ L_2}= \{ w_1\circ w_2\mid w_1\in
    L_1\text{ og } w_2\in L_2\}$

    -- sproget med alle de ord der er \alert{sammensætninger} af et ord
    fra $L_1$ efterfulgt af et ord fra $L_2$
  \item stjernen $\alert{L_1^*}= \{ w_1\circ w_2\circ\dots\circ w_k\mid
    \text{alle } w_i\in L_1\}$

    -- sproget med alle de ord der er \alert{sammensætninger} af
    \alert{vilkårligt mange} ord fra $L_1$

    -- indeholder $\epsilon$: det tomme ord $=$ sammensætningen af
    \alert{$0$} ord fra $L_1$ \dots
  \end{itemize}
\end{frame}

\section[Regulære udtryk]{Regulære udtryk igen}

\begin{frame}
  Vi kan beskrive sprog som mængder: (her lader vi $\Sigma= \{ a, b\}$ )
  \begin{itemize}
  \item $L_1= \{ a, b, ab\}$ \quad (et \emph{endeligt} sprog)
  \item $L_2= \{ a^n\mid n\in \Nat\}$ \quad-- alle ord der indeholder kun
    $a$, af vilkårlig længde
  \item $L_3= \{ a^n b a^m\mid n, m\in \Nat\}$ ~-- alle ord der
    indeholder præcist ét $b$
  \item $L_4= \{ a^n b^n\}$ \quad-- alle ord der indeholder et antal $a$
    og så \emph{samme} antal $b$
  \end{itemize}
  eller ved hjælp af \alert{regulære udtryk}:
  \begin{itemize}
  \item $L_1= a\cup b\cup ab$
  \item $L_2= a^*$
  \item $L_3= a^*\circ b\circ a^*$
  \item $L_4=$ ???

    (vi skal senere se at $L_4$ \emph{ikke} kan beskrives ved regulære
    udtryk!)
  \end{itemize}
\end{frame}

\begin{frame}[t]
  \alert{Definition 1.52:} Et \alert{regulært udtryk} over et alfabet
  $\Sigma$ er et udtryk af formen
  \begin{enumerate}
  \item\label{1.52:1} $a$ for et $a\in \Sigma$,
  \item $\epsilon$,
  \item\label{1.52:3} $\emptyset$,
  \item\label{1.52:4} $(R_1\cup R_2)$, hvor $R_1$ og $R_2$ er regulære udtryk,
  \item $(R_1\circ R_2)$, hvor $R_1$ og $R_2$ er regulære udtryk, eller
  \item\label{1.52:6} $(R_1^*)$, hvor $R_1$ er et regulært udtryk.
  \end{enumerate}

  \medskip
  \only<1>{%
    -- en \alert{induktiv} (eller \alert{rekursiv}) definition:
    \ref{1.52:1}. til \ref{1.52:3}. giver de basale \emph{byggesten}, og
    \ref{1.52:4}. til \ref{1.52:6}. giver \emph{byggeregler} til hvordan
    man kan sætte ting sammen.

    -- parenteserne udelades ofte

  }%
  \only<2>{%
    Eksempler (med $\Sigma= \{ a, b\}$ ):

    $a$, $b$, $a\cup b$, $(a\cup b)^*$, $(a\cup b)^*\circ b$, $((a\cup
    b)^*\circ b)^*$
  }
\end{frame}

\begin{frame}[t]
  \alert{Definition 1.52, fortsat:} \alert{Sproget}, som et regulært udtryk
  $R$ beskriver, betegnes $\llbracket R\rrbracket$ og er defineret som
  følger:
  \begin{enumerate}
  \item $\llbracket a\rrbracket= \{ a\}$
  \item $\llbracket \epsilon\rrbracket= \{ \epsilon\}$
  \item $\llbracket \emptyset\rrbracket= \emptyset$
  \item $\llbracket R_1\cup R_2\rrbracket= \llbracket R_1\rrbracket\cup
    \llbracket R_2\rrbracket$
  \item $\llbracket R_1\circ R_2\rrbracket= \llbracket R_1\rrbracket\circ
    \llbracket R_2\rrbracket$
  \item $\llbracket R_1^*\rrbracket= \llbracket R_1\rrbracket^*$
  \end{enumerate}

  \medskip \only<2>{%
    -- Sipser skriver $L( R)$ i stedet for $\llbracket R\rrbracket$. Jeg
    vil bruge begge notationer

    \medskip \alert{Udvidelser:}
    \begin{itemize}
    \item $\Sigma= a_1\cup a_2\cup\dots\cup a_n$ (hvis sproget er $\Sigma=
      \{ a_1, a_2,\dots, a_n\}$ )
    \item $R^+= R\circ R^*$
    \end{itemize}
  }
\end{frame}

\begin{frame}
  \alert{Eksempler} (1.53): (for $\Sigma= \{ 0, 1\}$ )
  \begin{enumerate}
  \item $\llbracket 0^* 1 0^*\rrbracket=$ sproget med alle ord der
    indeholder symbolet $1$ \emph{præcist} én gang
  \item $\llbracket \Sigma^* 1 \Sigma^*\rrbracket=$ sproget med alle ord
    der indeholder symbolet $1$ \emph{mindst} én gang
    \setcounter{enumi}3
  \item $\llbracket ( 0 1^+)^*\rrbracket=$ sproget af alle ord hvori
    ethvert $0$ efterfølges af mindst ét $1$
  \item $\llbracket ( \Sigma \Sigma)^*\rrbracket= \{ w\mid |w| \text{ er
      et lige tal} \}$
    \setcounter{enumi}7
  \item $\llbracket 0 \Sigma^* 0\cup 1 \Sigma^* 1\cup 0\cup 1\rrbracket=\{
    w\mid \text{start- og slutsymbolet i $w$ er ens}\}$
  \item $( 0\cup \epsilon) 1^*= 01^*\cup 1^*$
    \renewcommand{\theenumi}X
  \item $( 0\cup 1)^*=( 0^* 1^*)^*$
  \end{enumerate}
\end{frame}

\section{Regulære sprog}

\begin{frame}
  \alert{Definition:} Et sprog kaldes \alert{regulært} hvis det kan
  frembringes af et regulært udtryk.
  \begin{itemize}
  \item eller måske: Et sprog $L\subseteq \Sigma^*$, for $\Sigma$ et
    alfabet, siges at være \alert{regulært} hvis der findes et
    regulært udtryk $R$ over $\Sigma$ for hvilket $L=\llbracket
    R\rrbracket$.
  \end{itemize}

  \medskip \alert{Spørgsmål} der trænger sig på:
  \begin{itemize}
  \item Er alle sprog regulære? \hfill\blue{Nej.  Se afsnit 1.4
      (lektion 4)}
  \item Findes der andre måder at frembringe regulære sprog på? \\
    \hfill\blue{Ja.  F.x.\ endelige automater; se afsnit 1.1 (lektion
      2)}
  \item Findes der også måder at frembringe ikke-regulære sprog på? \\
    \hfill\blue{Ja.  F.x.\ kontekstfrie grammatikker; se afsnit 2.1
      (lektion 5)}
  \item Kan alle sprog så beskrives vha.\ disse metoder? \\
    \hfill\blue{Nej, langt fra.  Se afsnit 2.3 (lektion 7)}
  \item Hvad gør vi så?  \hfill\blue{Venter på B\&K-kurset næste
      semester}
  \end{itemize}

\end{frame}
